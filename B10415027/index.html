ch4 Locking

xv6是要運作在多核心處理器上的，多個處理器會同時在運作，但這些處理器卻是共享著同一個記憶體，可能會發生超在或是分配不均的情況，所以lock就是非常重要的一環，它提供了互斥，可以確保在一個時間只有一個cpu存取資料，防止相互的干擾，保護數據結構的平衡，也維持資料的正確性和完整。

如下程式，xv6主要使用spinlock以pushcli及popcli方法去紀錄與控制是否允許中斷處理,因為一般情況下內核對數據結構的佔用時間一般比較短,使用spinlock 能夠實現和減少處理同步時容易出現的deadlock問題。

// spinlock結構
struct spinlock {
  uint locked;//是否上鎖

  // Debug
  char *name;//鎖的名稱
  struct cpu *cpu;// cpu結構
  uint pcs[10];//程式計數器（堆疊形式）
  
};

struct list{
    int data;
    struct list *next;
};

struct list *list = 0;

void
insert(int data)
{
    struct list *l;
    l = malloc(sizeof *l);
    l->data = data;
    l->next = list;
    list = l;
}

但如上段程式，如果兩個處理器同時執行insert，在兩個執行完l->next = list; 進入下一行後進入的就會蓋掉前面的，因為next已經被設為list，這就是競爭關係(Race Condition)

為了避免此狀況，所以需要用到locking機制，lock會提供互斥，確保正常運作，

struct list *list = 0;
struct lock listlock;

void
insert(int data)
{
  struct list *l;

  acquire(&listlock);
  l = malloc(sizeof *l);
  l->data = data;
  l->next = list;
  list = l;
  release(&listlock);
}

如上面這段程式中，利用acquire和release，所製造出的Critical Section，能使資料結構不會被改變，另一個處理器必須等到前一個存取完後才可運行，這就是Locks的互斥概念。

而為了不讓race發生，我們必須用xchg（如下1.程式所示）來確定是否鎖上，確保atomic操作，xchg交換一個寄存器和內存字的值,acquire(如下2.程式所示）通過循環反复xchg,每一次執行 lk->locked 然后設為1，如果有lock了，lk->locked就已經為1，所以xchg返回的內存字值為1,繼續loop,如果xchg傳回0,但acquire已成功lock,代表locked從0變成了1，這時候就可以跳出loop,acquire會記錄獲得lock的處理器及訊息，便於之後debug，當process獲得lock卻沒釋放，便能讀取並保護某個address,若無法拿到的話就不停地等待,直到lock的使用權到手，而release則是相反，用於清除訊息並釋放lock。

1.//xchg//
static inline uint xchg(volatile uint *addr, uint newval)
{
uint result;

// The + in "+m" denotes a read−modify−write operand.
asm volatile("lock; xchgl %0, %1" :
"+m" (*addr), "=a" (result) :
"1" (newval) :
"cc");
return result;
}

acquire和release操作來代表獲得鎖和解除鎖的佔用。

2.// 初始化lock
void
initlock(struct spinlock *lk, char *name)
{
  // 初始化
  lk−>name = name;
  lk−>locked = 0;
  lk−>cpu = 0;
}

// 獲得lock
void
acquire(struct spinlock *lk)
{
  // 執行不允許中斷，目的是避免deadlock，並加入堆疊紀錄
  pushcli();
  // 若lock正被其他處理器佔用
  if(holding(lk))
    // 中斷流程，並回傳
    panic("acquire");

  // 用迭代確認結構變數lk的locked成員值是否為1，確保atomic
  // 若回傳1代表lock正被CPU佔用，則繼續跑迴圈做等待
  // 若回傳0則代表可以進行acquire，並將值設為1
  while(xchg(&lk−>locked, 1) != 0)
    ;

  // 在Compiler做編譯時有可能因為最佳化而改變了Assembly的順序
  // 此為Memory Barrier的技術，目的是確保順序性
  __sync_synchronize();

  // Debug使用
  // 將結構變數lk的cpu成員值設為當前cpu
  lk−>cpu = mycpu();
  // 紀錄程式計數器
  getcallerpcs(&lk, lk−>pcs);
}

// 釋放lock
void
release(struct spinlock *lk)
{
  // 若lock未被佔用
  if(!holding(lk))
    // 中斷流程，並回傳訊息
    panic("release");

  // 復原為初值
  lk−>pcs[0] = 0;
  lk−>cpu = 0;

  // Memory Barrier，用以確保順序性
  __sync_synchronize();

  // 將結構變數lk的locked成員值指定為0
  asm volatile("movl $0, %0" : "+m" (lk−>locked) : );

  // 執行允許中斷，並取出堆疊紀錄
  popcli();
}

而還有一點就是在處理程式中斷時可能會與其他執行的程式發生衝突發生deadlock，所以還需要以下兩個程式來協助

// 加入堆疊
void
pushcli(void)
{
  // 宣告紀錄是否中斷的變數
  int eflags;

  // 取得現在狀態
  eflags = readeflags();
  // 執行中斷禁止
  cli();
  // 結構變數cpu的ncli成員，代表堆疊的depth
  // 若depth＝0
  if(mycpu()−>ncli == 0)
    // FL_IF為一常數，代表可中斷
    // 結構變數cpu的intena成員，代表在加入堆疊前，是否可中斷
    // 將intena的值設為可中斷
    mycpu()−>intena = eflags & FL_IF;
  // 堆疊depth＋1
  mycpu()−>ncli += 1;
}

執行acquire的時候執行pushcli()取得中斷狀態並關閉中斷cli()，在mycpu()->ineta執行時紀錄狀態，為了CPU在取得複數的spinlock後又release時誤開啟中斷，因此又加入一個計數器，acquire 一定要在可能獲得鎖的 xchg 之前使用 pushcli，等於是上了兩道鎖，只有當CPU沒有取得任何spinlock時才恢復先前中斷觸發的狀態。

// 取出堆疊
void
popcli(void)
{
  // 取出當前狀態，並判斷是否已可中斷
  if(readeflags()&FL_IF)
    // 中斷流程，並回傳訊息
    panic("popcli − interruptible");
  // 堆疊depth減一
  // 若depth<0，代表堆疊已空
  if(−−mycpu()−>ncli < 0)
    // 中斷，並回傳
    panic("popcli");
  // 若堆疊depth=0，且為可中斷
  if(mycpu()−>ncli == 0 && mycpu()−>intena)
    // 執行中斷
    sti();
}

在執行release的時候執行popcli()去恢復中斷狀態，release 也一定要在釋放鎖的 xchg 之後去使用 popcli。如果計數器(ncli)顯示0代表CPU目前並沒有取得任何spinlock，此時即可恢復先前的中斷狀態intena。
