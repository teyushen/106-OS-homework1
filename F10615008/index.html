作業系統報告主題：Ch2 Page tables

一、前言

Page table是作業系統對虛擬地址（virtual address）和實際的物理地址進行轉換的一個table。通過Page table可以實現memory的擴充。

二、Paging handware

1.作用

x86的Page tables硬體連接兩個地址，將每一個虛擬地址映射成一個實際的物理地址。x86的page table從邏輯上包含2^20個PTE，每一個PTE包含一個20bit的PPN和一些標誌。Paging hardware通過Page table的前20bit來轉換虛擬地址，將其作為索引找到PTE，并用PTE中PPN替換地址的前20bit。然後page handware 賦值虛擬地址的低20位到實際的物理地址。這樣Page table實現了作業系統控制的以4096字節為對齊塊的虛擬地址到物理地址的轉換，這樣的一個塊我們稱之為“page”。

2.轉換過程

事實上轉換是分兩步完成的。一個page table存儲在記憶體中是一個two-level tree的結構。Paging handware使用虛擬地址的前10位選擇一個page directory進入。如果這個directory存在，paging handware就用接下來的10位從進入的page directory所對應的page table中選擇一個PTE；如果這個directory不存在，paging handware就拋出一個錯誤。這樣的兩層結構使得page table能夠忽略大範圍的虛擬地址沒有映射的情況。

3.標誌位的含義

每一個PTE所擁有的標誌位告訴paging handware與之關聯的虛擬地址應該怎樣用。PTE_P表明這個PTE是否在用，如果沒有設置就會拋出一個錯誤（表示沒有在用）。PTE_W決定是否可以發出指令寫入這個page，如果沒有設置表示只能夠讀取這個page。PTE_U控制用戶程式是否可以使用這個page，如果是0表示只有kernel能夠使用這個page。

三、進程地址空間

由entry創建的page table有足夠的映射允許kernel的C代碼去運行。然而，main馬上通過調用kvmalloc來改變成新的page table，因為有更加複雜的計劃去描述進程地址空間。
每一個進程都有一個單獨的page table，xv6告訴page table handware當進程切換的時候進行page table的切換。
當一個進程向xv6要更多的記憶體的時候，xv6首先找到空閒的物理page去提供存儲，然後添加新的PTEs到進程的page table指向新的物理page。xv6在這個PTEs中設置好PTE_U, PTE_W, 和 PTE_P這些參數。大多數進步不會用掉整個用戶地址空間，xv6對沒有用到的PTEs的PTE_P標誌清空。不同的進程page table將用戶地址轉換成不同的物理記憶體的page，所以每一個進程都有私人的用戶記憶體。
xv6包含kernel在每一個進程的page talbe中的所有映射。這樣映射的一個原因是kernel可以使用他自己的指令的數據。另外一個原因是kernel有時候需要能夠對給出的物理記憶體的page進行寫入。這樣的缺點是xv6不能夠使用超過2GB的物理記憶體。
在系統調用和中斷的時候切換user mode到kernel moded的時候擁有每一個進程所包含用戶記憶體和整個kernel的映射的page table是非常方便的，因為這樣的切換是不需要page table的切換的。大部分時候kernel沒有他自己的page table，他幾乎都是用其他進程的page table。
xv6確保每一個進程都有自己的記憶體並且每一個進程都看做擁有從零開始的虛擬記憶體即使物理記憶體不連續。xv6通過設置與進程相關的記憶體的虛擬地址的PETs中的PTE_U來實現第一步，然後通過page table將連續的虛擬地址轉換成和進程相關的物理page來實現第二步。

四、Code: creating an address space

程式流程

代碼解釋

// 創建并轉換一個page table
void kvmalloc(void)
{
  kpgdir = setupkvm();
  switchkvm();
}

// 設置kernel部分的page table
pde_t* setupkvm(void)
{
  pde_t *pgdir;
  struct kmap *k;

  if((pgdir = (pde_t*)kalloc()) == 0)
    return 0;
  memset(pgdir, 0, PGSIZE);           //分配一個page的記憶體來存儲page directory的東西
  if (P2V(PHYSTOP) > (void*)DEVSPACE)
    panic("PHYSTOP too high");
  for(k = kmap; k < &kmap[NELEM(kmap)]; k++)
    if(mappages(pgdir, k->virt, k->phys_end - k->phys_start,     //調用mappages創建kernel需要的映射
                (uint)k->phys_start, k->perm) < 0) {
      freevm(pgdir);
      return 0;
    }
  return pgdir;
}

// 創建kernel需要的映射
static int mappages(pde_t *pgdir, void *va, uint size, uint pa, int perm)
{
  char *a, *last;
  pte_t *pte;

  a = (char*)PGROUNDDOWN((uint)va);
  last = (char*)PGROUNDDOWN(((uint)va) + size - 1);
  for(;;){
    if((pte = walkpgdir(pgdir, a, 1)) == 0)  // 調用walkpgdir來查找在page table 中的PTE的地址
      return -1;
    if(*pte & PTE_P)
      panic("remap");
    *pte = pa | perm | PTE_P;          //初始化PTE標誌，保存相關頁碼
    if(a == last)
      break;
    a += PGSIZE;
    pa += PGSIZE;
  }
  return 0;
}

// 查找在page table 中的PTE的地址，模擬x86paging handware查找PTE的虛擬地址
static pte_t * walkpgdir(pde_t *pgdir, const void *va, int alloc)
{
  pde_t *pde;
  pte_t *pgtab;

  pde = &pgdir[PDX(va)];    // 使用前10bit來找到page directory
  if(*pde & PTE_P){    
    pgtab = (pte_t*)P2V(PTE_ADDR(*pde));   
  } else {
    if(!alloc || (pgtab = (pte_t*)kalloc()) == 0)      
      return 0;
    memset(pgtab, 0, PGSIZE);
    *pde = V2P(pgtab) | PTE_P | PTE_W | PTE_U;
  }
  return &pgtab[PTX(va)];   // 通過后10bit來確定地址
}

五、物理記憶體的分配

kernel必須為page table、進程用戶記憶體、kernel stacks和pipe buffers在運行時分配和釋放記憶體。
xv6在PHYSTOP和kernel結束的這段時間裡面使用記憶體，它通常一次分配或者釋放4096字節的page。它通過pages之間的鏈接來判斷哪些鏈接是可以使用的。分配記憶體包含從鏈接表中移除一個page；釋放記憶體包含添加一個空page到這個鏈接表中
有一個根本問題：為了初始化空鏈錶所有的記憶體必須被映射，但是創建一個含有這些映射的page table涉及到為page table分配一個page。xv6通過在實體間相互獨立的page分配器來解決這個問題，使得剛好在kernel結束的時候分配記憶體。這個分配器不支持釋放記憶體並且只是限制在4MB大小，但是足夠為kernel的第一個page table分配記憶體。

六、Code: Physical memory allocator

程式流程

代碼解釋

void kinit1(void *vstart, void *vend)
{
  initlock(&kmem.lock, "kmem");   //初始化鎖
  kmem.use_lock = 0;
  freerange(vstart, vend);        // add memory to the free list via per-page
}

void kinit2(void *vstart, void *vend)
{
  freerange(vstart, vend);
  kmem.use_lock = 1;
}

void freerange(void *vstart, void *vend)
{
  char *p;
  p = (char*)PGROUNDUP((uint)vstart);
  for(; p + PGSIZE <= (char*)vend; p += PGSIZE)
    kfree(p);                    
}

void kfree(char *v)
{
  struct run *r;

  if((uint)v % PGSIZE || v < end || V2P(v) >= PHYSTOP)
    panic("kfree");

  memset(v, 1, PGSIZE);

  if(kmem.use_lock)
    acquire(&kmem.lock);
  r = (struct run*)v;
  r->next = kmem.freelist;
  kmem.freelist = r;
  if(kmem.use_lock)
    release(&kmem.lock);
}

七、地址空間的用戶部分

每一個用戶進程都從地址0開始，地址空間的底部包含了程式的文本、數據和棧。堆在棧的上面以保證當進程調用sbrk的時候能夠擴展。注意：文本、數據和堆棧部分在進程的地址空間是連續分佈的，但是xv6可以自由為這些部分使用非連續的物理page。
棧是一個單一的page並且出現在由exec創建的初始化的部分裡面。字串包含命令行參數和一個指向他們的指針數組在棧的頂端。在他們下方的是程式調用main所必須的部分。為了防止在堆棧中生成新的堆棧頁面，xv6在堆棧的低端放置了一個保護頁面。保護page不複雜映射，這樣如果棧運行在棧的頁面，因為無法轉化一個錯誤的地址所以硬件會生成一個錯誤。現實生活中的作業系統可能為棧分配更多的空間使得其能夠在一個頁面下繼續使用。

八、Code: sbrk

程式流程

代碼解釋

// 為進程增加或者縮小記憶體
int growproc(int n)
{
  uint sz;
  struct proc *curproc = myproc();

  sz = curproc->sz;
  if(n > 0){
    if((sz = allocuvm(curproc->pgdir, sz, sz + n)) == 0)   //對page進行更新
      return -1;
  } else if(n < 0){
    if((sz = deallocuvm(curproc->pgdir, sz, sz + n)) == 0)  //對page進行更新
      return -1;
  }
  curproc->sz = sz;
  switchuvm(curproc);
  return 0;
}

九、Code: exec

程式流程

代碼解釋

int exec(char *path, char **argv)
{
  char *s, *last;
  int i, off;
  uint argc, sz, sp, ustack[3+MAXARG+1];
  struct elfhdr elf;
  struct inode *ip;
  struct proghdr ph;
  pde_t *pgdir, *oldpgdir;
  struct proc *curproc = myproc();

  begin_op();

  if((ip = namei(path)) == 0){    //打開相應路勁的文件
    end_op();
    cprintf("exec: fail\n");
    return -1;
  }
  ilock(ip);
  pgdir = 0;

  // 檢車ELF的頭部信息
  if(readi(ip, (char*)&elf, 0, sizeof(elf)) != sizeof(elf))
    goto bad;
  if(elf.magic != ELF_MAGIC)
    goto bad;

  if((pgdir = setupkvm()) == 0)
    goto bad;

  // 加載程式到記憶體
  sz = 0;
  for(i=0, off=elf.phoff; i<elf.phnum; i++, off+=sizeof(ph)){
    if(readi(ip, (char*)&ph, off, sizeof(ph)) != sizeof(ph))
      goto bad;
    if(ph.type != ELF_PROG_LOAD)
      continue;
    if(ph.memsz < ph.filesz)
      goto bad;
    if(ph.vaddr + ph.memsz < ph.vaddr)
      goto bad;
    if((sz = allocuvm(pgdir, sz, ph.vaddr + ph.memsz)) == 0)
      goto bad;
    if(ph.vaddr % PGSIZE != 0)
      goto bad;
    if(loaduvm(pgdir, (char*)ph.vaddr, ip, ph.off, ph.filesz) < 0)
      goto bad;
  }
  iunlockput(ip);
  end_op();
  ip = 0;

  // 在下一個page的分界線分配兩個page
  sz = PGROUNDUP(sz);
  if((sz = allocuvm(pgdir, sz, sz + 2*PGSIZE)) == 0)
    goto bad;
  clearpteu(pgdir, (char*)(sz - 2*PGSIZE));
  sp = sz;

  // 加載其他的參數進入棧中
  for(argc = 0; argv[argc]; argc++) {
    if(argc >= MAXARG)
      goto bad;
    sp = (sp - (strlen(argv[argc]) + 1)) & ~3;
    if(copyout(pgdir, sp, argv[argc], strlen(argv[argc]) + 1) < 0)    // copyout將參數賦值到棧中取的時候將會檢查目標page是否可讀
      goto bad;
    ustack[3+argc] = sp;
  }
  ustack[3+argc] = 0;

  ustack[0] = 0xffffffff;  // fake return PC
  ustack[1] = argc;
  ustack[2] = sp - (argc+1)*4;  // argv pointer

  sp -= (3+argc+1) * 4;
  if(copyout(pgdir, sp, ustack, (3+argc+1)*4) < 0)
    goto bad;

  // 為調試存儲程式的名稱
  for(last=s=path; *s; s++)
    if(*s == '/')
      last = s+1;
  safestrcpy(curproc->name, last, sizeof(curproc->name));
  
  oldpgdir = curproc->pgdir;
  curproc->pgdir = pgdir;
  curproc->sz = sz;
  curproc->tf->eip = elf.entry;  // main
  curproc->tf->esp = sp;
  switchuvm(curproc);
  freevm(oldpgdir);
  return 0;

 bad:
  if(pgdir)
    freevm(pgdir);
  if(ip){
    iunlockput(ip);
    end_op();
  }
  return -1;
}

struct elfhdr {
  uint magic;  // must equal ELF_MAGIC
  uchar elf[12];
  ushort type;
  ushort machine;
  uint version;
  uint entry;
  uint phoff;
  uint shoff;
  uint flags;
  ushort ehsize;
  ushort phentsize;
  ushort phnum;
  ushort shentsize;
  ushort shnum;
  ushort shstrndx;
};

struct proghdr {

  uint off;
  uint vaddr;
  uint paddr;
  uint filesz;
  uint memsz;
  uint flags;
  uint align;
};
